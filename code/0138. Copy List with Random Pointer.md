```java
/*
// Definition for a Node.
class Node {
    int val;
    Node next;
    Node random;

    public Node(int val) {
        this.val = val;
        this.next = null;
        this.random = null;
    }
}
*/
class Solution {
    public Node copyRandomList(Node head) {
        Node cur = head;
        while(head != null){
            Node node = new Node(head.val);
            node.next = head.next;
            head.next = node;
            head = head.next.next;
        }
        Node ol = cur;
        while(ol!=null){
            Node ne = ol.next;
            if(ol.random != null){
                ne.random = ol.random.next;
            }
            ol = ne.next;
        }
        Node dummy = new Node(-1);
        Node cur2 = dummy;
        while(cur!=null){
            cur2.next = cur.next;
            cur2 = cur2.next;
            cur.next = cur2.next;
            cur = cur.next;
        }
        return dummy.next;
    }
}
```
